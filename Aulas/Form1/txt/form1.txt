create database db_Mercado;

use db_Mercado;

create table produto (
id int auto_increment,
codigo varchar(150) NOT NULL,
descricao varchar(300),
p_compra varchar(150) NOT NULL,
p_venda varchar(150) NOT NULL,
estoque varchar(150) NOT NULL,
primary key(id));

create table funcionario (
id int auto_increment,
matricula varchar(150) NOT NULL,
nome varchar(300) NOT NULL,
logradouro varchar(150) NOT NULL,
bairro varchar(150) NOT NULL,
cidade varchar(150) NOT NULL,
telefone varchar(15) NOT NULL,
admissao varchar(150) NOT NULL,
demissao varchar(150),
setor varchar(150) NOT NULL,
primary key(id));

create table usuario (
id int auto_increment,
codigo varchar(150) NOT NULL,
nome varchar(300) NOT NULL,
nome_usuario varchar(150) NOT NULL,
senha varchar(150) NOT NULL,
primary key(id));

create table fornecedor (
id int auto_increment,
codigo varchar(150) NOT NULL,
cpf varchar(300),
cnpj varchar(150),
ie varchar(150) NOT NULL,
logradouro varchar(150) NOT NULL,
bairro varchar(15) NOT NULL,
cidade varchar(150) NOT NULL,
telefone varchar(150) NOT NULL,
primary key(id));
